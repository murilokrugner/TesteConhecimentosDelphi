CREATE DATABASE DELPHIDB

CREATE TABLE CLIENTES(CODIGO INT NOT NULL PRIMARY KEY AUTO_INCREMENT, NOME VARCHAR(60) NOT NULL, CIDADE VARCHAR(40), UF VARCHAR(2), INDEX(CODIGO)); 

CREATE TABLE PRODUTOS(CODIGO INT PRIMARY KEY NOT NULL AUTO_INCREMENT, DESCRICAO VARCHAR(60) NOT NULL, PRECO DECIMAL(15,6), INDEX(CODIGO)); 

CREATE TABLE VENDAS(CODIGO INT NOT NULL PRIMARY KEY auto_increment, COD_VENDA INT NOT NULL, DATA_EMISSAO DATE NOT NULL, COD_CLIENTE INT, TOTAL DECIMAL(15,6),
FOREIGN KEY(COD_CLIENTE) REFERENCES CLIENTES(CODIGO), INDEX(CODIGO));

CREATE TABLE VENDAS_PRODUTOS(CODIGO INT NOT NULL PRIMARY KEY AUTO_INCREMENT, COD_VENDA INT NOT NULL, 
COD_PRODUTO INT NOT NULL, QUANTIDADE INT, VALOR_UNITARIO DECIMAL(15,6), VALOR_TOTAL DECIMAL(15,6), 
FOREIGN KEY (COD_VENDA) REFERENCES VENDAS(CODIGO),
FOREIGN KEY (COD_PRODUTO) REFERENCES PRODUTOS(CODIGO),
INDEX(CODIGO));
